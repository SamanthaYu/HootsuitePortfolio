public void select(int x, int y) {  // Selects menu items
    // If mouse selects a color
    if (redRect.inside(x, y)) { 
        numColor = 1;
    }
    else if (orangeRect.inside(x, y)) {
        numColor = 2;
    }
    else if (yellowRect.inside(x, y)) {
        numColor = 3;
    }
    else if (greenRect.inside(x, y)) {
        numColor = 4;
    }
    else if (blueRect.inside(x, y)) {
        numColor = 5;
    }
}
